func (b *basicSubject) Release(errored bool) gomerr.Gomerr {
	errors := make([]gomerr.Gomerr, 0)

		ge := principal.Release(errored)
		if ge != nil {
			errors = append(errors, ge)
		}
	}

	if len(errors) > 0 {
		return gomerr.Batcher(errors)
	}

	return nil
}
